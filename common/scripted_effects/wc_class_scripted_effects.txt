		limit = {

		}
		ai_add_priest_class_effect = yes
	}
	else_if = {

	}
	else_if = {
		limit = { shamanic_public_religion_trigger = yes }

	}

		limit = { religion_group = fel_group }
		ai_add_warlock_class_effect = yes
	}
	else_if = {
		limit = { religion_group = loa_group }
		ai_add_loa_priest_class_effect = yes
	}
	else_if = {
		limit = { religion_group = necromancy_group }

	}
	else_if = {
		limit = { religion_group = void_group }

		limit = { religion_group = pandaren_religion_group }

	}
