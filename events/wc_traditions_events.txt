### Shattered Hand clan ###

# Blade hand
character_event = {
	id = WCTRD.200
	desc = EVTDESC_WCTRD_200
	picture = GFX_evt_makgora
	border = GFX_event_normal_frame_war

	is_triggered_only = yes
	
	has_dlc = "Reapers"

	only_playable = yes
	only_capable = yes
	only_men = yes

	trigger = {
		is_alive = yes
		is_dying = no
		
		NOT = { trait = one_handed }
		culture = shattered_hand
		liege = { culture = shattered_hand }
	}

	option = {
		name = EVTOPTA_WCTRD_200
		add_only_trait_one_handed_effect = yes
		if = {
			limit = {
				NOR = {
					trait = brave 
					trait = craven
				}
			}
			random_list = {
				75 = {}
				25 = { add_trait_brave_effect = yes }
			}
		}
		remove_trait_craven_effect = yes
		add_artifact = prosthetic_blade_hand

		ai_chance = {
			factor = 95
		}
	}

	option = {
		name = I_CANT
		banish = yes
		add_trait_craven_effect = yes
		prestige = -100
		any_character = {
			limit = {
				culture = shattered_hand
				trait = one_handed
			}
			opinion = { modifier = opinion_traitor who = ROOT years = 1000 }
		}

		ai_chance = {
			factor = 5
			modifier = {
				factor = 40
				trait = craven
			}
		}
	}
}
