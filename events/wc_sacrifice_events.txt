								}

								clear_event_target = target_sacrificed_character

							}
						}
					}
				}
			}
		}

		if = { # Get Offmap Old Gods Currency if worship Old Gods
			limit = { publicly_worship_one_of_old_gods_trigger = yes }

			trigger_switch = {
				on_trigger = has_character_modifier
