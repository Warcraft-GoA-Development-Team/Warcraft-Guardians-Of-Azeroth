			}

		}

			remove_building = ca_old_god_2
		}
		offmap_cthun = {

		}
	}

	option = {
		name = EVTOPTA_WCOME.200

            limit = {
                is_female = yes
            }
            create_bloodline = {
                type = godslayer_cthun
                inheritance = matrilineal
            }
        }
        else = {
            create_bloodline = {
                type = godslayer_cthun
            }
        }
		
		any_player = {
			narrative_event = { id = WCOME.201 }
		}
	}
}
