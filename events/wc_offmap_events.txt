					else = {
						add_offmap_currency = { offmap = offmap_cthun value = 50 }
					}

						limit = {
							any_owned_bloodline = { has_bloodline_flag = bloodline_insanity_gain }
						}
						add_offmap_currency = { offmap = offmap_cthun value = 30 }
