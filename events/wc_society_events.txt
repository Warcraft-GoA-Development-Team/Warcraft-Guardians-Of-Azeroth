#Write down whispers
character_event = {
	id = WCCSO.10200
	desc = EVTDESC_WCCSO.10200
	picture = GFX_evt_madness
    border = GFX_event_normal_frame_religion

    is_triggered_only = yes

    trigger = { can_use_society_power_write_whispers_trigger = yes }

    option = {
    	name = EVTOPTA_WCCSO.10200
    	add_character_modifier = {
    		modifier = writing_whispers
			years = 2
    	}
		if = {
			limit = {
				NOT = { piety = 500 }
			}
			piety = -250
			set_character_flag = whispers_quality_1
		}
		else_if = {
			limit = {
				piety = 500
				NOT = { piety = 750 }
			}
			piety = -500
			set_character_flag = whispers_quality_2
		}
		else_if = {
			limit = {
				piety = 750
				NOT = { piety = 1000 }
			}
			piety = -750
			set_character_flag = whispers_quality_3
		}
		else = {
			piety = -1000
			set_character_flag = whispers_quality_4
		}
		character_event = { id = WCCSO.10201 days = 90 random = 60 }
	}
}
