			character_event = { id = WCCSO.6999 days = 3 }

			factor = 20
			modifier = {
				factor = 0
				trait = zealous
			}
			modifier = {
				factor = 1.5
				trait = trusting
			}
			modifier = {
				factor = 1.5
				is_dumb_trigger = yes
			}
			modifier = {
				factor = 1.5
				trait = hedonist
			}
			modifier = {
				factor = 1.5
				trait = lustful
			}
			modifier = {
				factor = 1.5
				trait = cynical

	option = {
		name = { text = EVTOPTB_WCCSO_7001 }

			event_target:secret_society_recruiter = {

			factor = 80

				factor = 1.5

	}
}
