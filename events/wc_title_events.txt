namespace = WCTLT

# on_character_convert_culture
# Changes title flag, name etc.
character_event = {
	id = WCTLT.100

	is_triggered_only = yes
	hide_window = yes
	
	only_playable = yes

	trigger = {
		OR = {
			e_quelthalas = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					holder_scope = { character = ROOT }
					trigger_if = {
						limit = { ROOT = { culture = blood_elf } }
						NOT = { has_title_flag = blood_flag }
					}
					trigger_else = {
						ROOT = { culture = high_elf }
						has_title_flag = blood_flag
					}
				}
			}
			k_quelthalas = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					trigger_if = {
						limit = { ROOT = { culture = blood_elf } }
						NOT = { has_title_flag = blood_flag }
					}
					trigger_else = {
						ROOT = { culture = high_elf }
						has_title_flag = blood_flag
					}
				}
			}
			e_lordaeron = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					can_change_forsaken_flag_trigger = yes
				}
			}
			k_lordaeron = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					can_change_forsaken_flag_trigger = yes
				}
			}
			d_lordaeron = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					can_change_forsaken_flag_trigger = yes
				}
			}
			c_lordaeron = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					can_change_forsaken_flag_trigger = yes
				}
			}
			k_gilneas = {
				trigger_if = {
					limit = { holder_scope = { character = ROOT } }
					trigger_if = {
						limit = { ROOT = { dynasty = 24114 } }
						NOT = { has_title_flag = aderician_flag }
					}
					trigger_else = {
						ROOT = { NOT = { dynasty = 24114 } }
						has_title_flag = aderician_flag
					}
				}
			}
			e_pandaria_empire = { holder_scope = { character = ROOT } }	#Changes name to Culture + Empire
		}
	}
	
	immediate = {
		e_quelthalas = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				if = {
					limit = {
						ROOT = { culture = blood_elf }
						NOT = { has_title_flag = blood_flag }
					}
					set_title_flag = blood_flag
					set_coa = d_sindorei
				}
				else_if = {
					limit = {
						ROOT = { culture = high_elf }
						has_title_flag = blood_flag
					}
					clr_title_flag = blood_flag
					reset_coa = THIS
				}
			}
		}
		k_quelthalas = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				if = {
					limit = {
						ROOT = { culture = blood_elf }
						NOT = { has_title_flag = blood_flag }
					}
					set_title_flag = blood_flag
					set_coa = d_sindorei
				}
				else_if = {
					limit = {
						ROOT = { culture = high_elf }
						has_title_flag = blood_flag
					}
					clr_title_flag = blood_flag
					reset_coa = THIS
				}
			}
		}
		e_lordaeron = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				change_forsaken_flag_effect = yes
			}
		}
		k_lordaeron = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				change_forsaken_flag_effect = yes
			}
		}
		d_lordaeron = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				change_undercity_flag_effect = yes
			}
		}
		c_lordaeron = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				change_undercity_flag_effect = yes
			}
		}
		k_gilneas = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				if = {
					limit = {
						ROOT = { dynasty = 24114 }
						NOT = { has_title_flag = aderician_flag }
					}
					set_title_flag = aderician_flag
					set_coa = d_gilneas_aderician
				}
				else_if = {
					limit = {
						ROOT = { NOT = { dynasty = 24114 } }
						has_title_flag = aderician_flag
					}
					clr_title_flag = aderician_flag
				}
			}
		}
		#Changes name to Culture + Empire
		e_pandaria_empire = {
			if = {
				limit = { holder_scope = { character = ROOT } }
				change_pandaria_name_effect = yes
			}
		}
	}
}

# on_new_holder
# on_new_holder_inheritance
# on_new_holder_usurpation
# Changes title flag, name etc.
character_event = {
	id = WCTLT.105

	is_triggered_only = yes
	hide_window = yes
	
	only_playable = yes

	trigger = {
		FROM = {
			trigger_if = {
				limit = {
					OR = {
						title = e_quelthalas
						title = k_quelthalas
					}
				}
				trigger_if = {
					limit = { ROOT = { culture = blood_elf } }
					NOT = { has_title_flag = blood_flag }
				}
				trigger_else = {
					ROOT = { culture = high_elf }
					has_title_flag = blood_flag
				}
			}
			trigger_else_if = {
				limit = {
					OR = {
						title = e_lordaeron
						title = k_lordaeron
						title = d_lordaeron
						title = c_lordaeron
					}
				}
				can_change_forsaken_flag_trigger = yes
			}
			trigger_else_if = {
				limit = {
					title = k_gilneas
				}
				trigger_if = {
					limit = { ROOT = { dynasty = 24114 } }
					NOT = { has_title_flag = aderician_flag }
				}
				trigger_else = {
					ROOT = { NOT = { dynasty = 24114 } }
					has_title_flag = aderician_flag
				}
			}
			trigger_else = {
				title = e_pandaria_empire	#Changes name to Culture + Empire
			}
		}
	}
	
	immediate = {
		FROM = {
			if = {
				limit = {
					OR = {
						title = e_quelthalas
						title = k_quelthalas
					}
				}
				if = {
					limit = {
						ROOT = { culture = blood_elf }
						NOT = { has_title_flag = blood_flag }
					}
					set_title_flag = blood_flag
					set_coa = d_sindorei
				}
				else = {
					clr_title_flag = blood_flag
					reset_coa = THIS
				}
			}
			else_if = {
				limit = {
					OR = {
						title = e_lordaeron
						title = k_lordaeron
					}
				}
				change_forsaken_flag_effect = yes
			}
			else_if = {
				limit = {
					OR = {
						title = d_lordaeron
						title = c_lordaeron
					}
				}
				change_undercity_flag_effect = yes
			}
			else_if = {
				limit = {
					OR = {
						title = k_gilneas
					}
				}
				if = {
					limit = {
						ROOT = { dynasty = 24114 }
						NOT = { has_title_flag = aderician_flag }
					}
					set_title_flag = aderician_flag
					set_coa = d_gilneas_aderician
				}
				else = {
					clr_title_flag = aderician_flag
				}
			}			
			else = {
				#Changes name to Culture + Empire
				change_pandaria_name_effect = yes
			}
		}
	}
}
